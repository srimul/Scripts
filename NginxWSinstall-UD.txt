Create EC2 instance and install Nginx Web server through User Data and start the Web server and access the web page from your browser  - done

#!/bin/bash
touch /tmp/file0
sudo apt update -y
touch /tmp/file1
sudo apt install nginx -y
sudo apt install lynx -y
sudo ufw allow 'Nginx HTTP'
git clone https://github.com/munireddy/indexhtml.git
cp indexhtml/index.html //var/www/html/
touch /tmp/file2.txt
cp indexhtml/index.html /tmp/index.html
